let state = {
  qIndex: 0,
  score: 0,
  streak: 0,
  correct: 0,
  wrong: 0,
  selected: new Set(),
  submitted: false,
  order: []
};

function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

function restart() {
  state = { qIndex: 0, score: 0, streak: 0, correct: 0, wrong: 0, selected: new Set(), submitted: false, order: shuffle([...Array(QUESTIONS.length).keys()]) };
  document.getElementById('score-screen').style.display = 'none';
  document.getElementById('quiz-area').style.display = 'block';
  updateStats();
  renderQuestion();
}

function currentQ() {
  return QUESTIONS[state.order[state.qIndex]];
}

function updateStats() {
  document.getElementById('stat-score').textContent = state.score;
  document.getElementById('stat-streak').textContent = state.streak;
  document.getElementById('stat-q').textContent = state.qIndex + 1;
  const pct = (state.qIndex / QUESTIONS.length) * 100;
  document.getElementById('progress').style.width = pct + '%';
}

function toggleCell(key) {
  if (state.submitted) return;
  const q = currentQ();
  if (q.ne.length === 0) return; // no pure NE — can't select cells

  if (state.selected.has(key)) {
    state.selected.delete(key);
  } else {
    state.selected.add(key);
  }
  renderCellStates();
}

function renderCellStates() {
  const q = currentQ();
  const rows = q.strategies.p1.length;
  const cols = q.strategies.p2.length;
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      const el = document.getElementById(`cell_${i}_${j}`);
      if (!el) continue;
      el.classList.remove('selected', 'correct', 'wrong', 'missed', 'disabled');
      const key = `${i},${j}`;
      if (state.submitted) {
        const isNE = q.ne.some(([r,c]) => r===i && c===j);
        const wasSelected = state.selected.has(key);
        if (isNE && wasSelected) el.classList.add('correct', 'disabled');
        else if (isNE && !wasSelected) el.classList.add('missed', 'disabled');
        else if (!isNE && wasSelected) el.classList.add('wrong', 'disabled');
        else el.classList.add('disabled');
      } else {
        if (state.selected.has(key)) el.classList.add('selected');
      }
    }
  }
}

  const neKeys = new Set(q.ne.map(([r,c]) => `${r},${c}`));
  const selArr = [...state.selected];
  const noNeBtn = document.getElementById('no-ne-btn');

  let isCorrect = false;

  if (q.ne.length === 0) {
    // Correct answer is "No Nash Equilibrium"
    isCorrect = noNeBtn && noNeBtn.dataset.selected === 'true';
  } else {
    // Check if selected set exactly matches NE set
    isCorrect = selArr.length === q.ne.length && selArr.every(k => neKeys.has(k));
  }

  // Partial credit: got some but not all
  const correctHits = selArr.filter(k => neKeys.has(k)).length;
  const wrongHits = selArr.filter(k => !neKeys.has(k)).length;
  const isPartial = !isCorrect && correctHits > 0 && wrongHits === 0 && q.ne.length > 0;

  const fb = document.getElementById('feedback');
  fb.style.display = 'block';

  if (isCorrect) {
    const pts = q.difficulty === 'easy' ? 10 : q.difficulty === 'medium' ? 20 : 30;
    state.score += pts + (state.streak >= 2 ? 5 : 0);
    state.streak++;
    state.correct++;
    fb.className = 'feedback correct-fb';
    const streakBonus = state.streak >= 2 ? ` (+5 streak bonus!)` : '';
    fb.innerHTML = `<span class="feedback-icon">✅</span><strong>Correct! +${pts} pts${streakBonus}</strong>${q.explanation}`;
  } else if (isPartial) {
    state.score += 5;
    state.streak = 0;
    fb.className = 'feedback partial-fb';
    fb.innerHTML = `<span class="feedback-icon">⚡</span><strong>Partial — you missed some NE cells. +5 pts</strong>${q.explanation}`;
  } else {
    state.streak = 0;
    state.wrong++;
    fb.className = 'feedback wrong-fb';
    fb.innerHTML = `<span class="feedback-icon">❌</span><strong>Not quite.</strong>${q.explanation}`;
  }

  renderCellStates();

  // Show NE badge on correct cells
  q.ne.forEach(([r,c]) => {
    const el = document.getElementById(`cell_${r}_${c}`);
    if (el) {
      let badge = el.querySelector('.cell-badge');
      if (!badge) { badge = document.createElement('div'); badge.className = 'cell-badge'; el.appendChild(badge); }
      badge.textContent = '★';
    }
  });

  if (noNeBtn) {
    noNeBtn.disabled = true;
    if (q.ne.length === 0) {
      noNeBtn.style.background = isCorrect ? 'var(--green)' : '#333';
      noNeBtn.style.color = isCorrect ? '#000' : 'var(--red)';
      noNeBtn.style.borderColor = isCorrect ? 'var(--green)' : 'var(--red)';
    }
  }

  document.getElementById('btn-submit').style.display = 'none';
  document.getElementById('btn-next').style.display = 'block';
  updateStats();
}

function next() {
  state.qIndex++;
  state.selected = new Set();
  state.submitted = false;

  if (state.qIndex >= QUESTIONS.length) {
    showScore();
  } else {
    updateStats();
    renderQuestion();
  }
}

function selectNoNE() {
  if (state.submitted) return;
  const btn = document.getElementById('no-ne-btn');
  const isSelected = btn.dataset.selected === 'true';
  btn.dataset.selected = isSelected ? 'false' : 'true';
  btn.style.background = !isSelected ? 'var(--accent)' : 'var(--surface2)';
  btn.style.color = !isSelected ? '#000' : 'var(--text)';
  btn.style.borderColor = !isSelected ? 'var(--accent)' : 'var(--border)';
}

function renderQuestion() {
  const q = currentQ();
  const rows = q.strategies.p1.length;
  const cols = q.strategies.p2.length;

  let html = `
    <div class="q-card">
      <div class="q-meta">
        <div class="q-num">Q${state.qIndex + 1} / ${QUESTIONS.length}</div>
        <div class="q-difficulty diff-${q.difficulty}">${q.difficulty}</div>
      </div>
      <div class="q-title">${q.title}</div>
      <div class="q-scenario">${q.scenario}</div>

      <div class="hint-box">
        <b>💡 How to read the matrix:</b> Each cell shows <span style="color:#74b9ff">P1's payoff</span> , <span style="color:#fd79a8">P2's payoff</span>.<br>
        Row player = Player 1 &nbsp;|&nbsp; Column player = Player 2
      </div>

      <div class="instruction">Click all Nash Equilibrium cells</div>

      <div class="legend">
        <div class="legend-item"><div class="legend-dot" style="background:#74b9ff"></div>Player 1 payoff</div>
        <div class="legend-item"><div class="legend-dot" style="background:#fd79a8"></div>Player 2 payoff</div>
        <div class="legend-item"><div class="legend-dot" style="background:var(--accent)"></div>Your selection</div>
      </div>

      <div class="matrix-wrap">
        <table class="matrix-table">
          <thead>
            <tr>
              <td class="axis-label"></td>
              <td class="axis-label" colspan="${cols}" style="text-align:center;color:var(--muted2)">← Player 2 →</td>
            </tr>
            <tr>
              <td class="axis"></td>
              ${q.strategies.p2.map(s => `<td class="axis">${s}</td>`).join('')}
            </tr>
          </thead>
          <tbody>`;

  for (let i = 0; i < rows; i++) {
    html += `<tr>`;
    if (i === Math.floor(rows/2)) {
      html += `<td class="axis" rowspan="${rows}" style="writing-mode:vertical-rl;transform:rotate(180deg);text-align:center;color:var(--muted2);padding:0 6px">↑ Player 1 ↑</td>`;
    } else if (i !== Math.floor(rows/2)) {
      if (rows === 2 && i === 0) html += `<td class="axis" rowspan="1" style="writing-mode:vertical-rl;transform:rotate(180deg);text-align:center;color:var(--muted2);padding:0 6px">↑ P1 ↑</td>`;
      else if (rows > 2) {} // handled by rowspan above
    }
    for (let j = 0; j < cols; j++) {
      html += `
        <td>
          <div class="pcell tooltip" id="cell_${i}_${j}" onclick="toggleCell('${i},${j}')">
            <div class="inner">
              <div class="payoffs">
                <span class="p1val">${q.payoffs.p1[i][j]}</span>
                <span class="sep"> , </span>
                <span class="p2val">${q.payoffs.p2[i][j]}</span>
              </div>
              <div class="coords">${q.strategies.p1[i]} / ${q.strategies.p2[j]}</div>
            </div>
            <span class="tooltip-text">P1: ${q.payoffs.p1[i][j]} &nbsp;|&nbsp; P2: ${q.payoffs.p2[i][j]}</span>
          </div>
        </td>`;
    }
    html += `</tr>`;
  }

  html += `</tbody></table></div>`;

  // No NE button
  html += `
    <div style="margin-bottom:16px">
      <button id="no-ne-btn"
        onclick="selectNoNE()"
        data-selected="false"
        style="background:var(--surface2);color:var(--text);border:1.5px solid var(--border);border-radius:10px;padding:10px 20px;font-family:'DM Sans',sans-serif;font-weight:700;font-size:13px;cursor:pointer;letter-spacing:0.5px;text-transform:uppercase;transition:all 0.18s;width:100%">
        ⊘ No Pure Strategy Nash Equilibrium
      </button>
    </div>`;

  // Hint
  html += `
    <details style="margin-bottom:16px">
      <summary style="font-family:'DM Mono',monospace;font-size:12px;color:var(--muted2);cursor:pointer;letter-spacing:1px;text-transform:uppercase;list-style:none">
        ▶ Show Hint
      </summary>
      <div class="hint-box" style="margin-top:10px;margin-bottom:0">${q.hint}</div>
    </details>`;

  // Feedback & buttons
  html += `
    <div class="feedback" id="feedback"></div>
    <div class="btn-row">
      <button class="btn btn-submit" id="btn-submit" onclick="submit()">Submit Answer</button>
      <button class="btn btn-next" id="btn-next" onclick="next()">${state.qIndex + 1 < QUESTIONS.length ? 'Next Question →' : 'See Results →'}</button>
    </div>
  </div>`;

  document.getElementById('quiz-area').innerHTML = html;

  // Fix row labels for non-2x2
  fixRowLabels(q, rows);
}

function fixRowLabels(q, rows) {
  // Re-render clean row labels
  const table = document.querySelector('.matrix-table tbody');
  if (!table) return;
  const trs = table.querySelectorAll('tr');
  // Remove auto-inserted vertical labels
  table.querySelectorAll('.axis[rowspan]').forEach(e => e.remove());
  trs.forEach((tr, i) => {
    // Insert row label cell at beginning
    const existing = tr.querySelector('.axis-row-label');
    if (!existing) {
      const td = document.createElement('td');
      td.className = 'axis axis-row-label';
      td.style.textAlign = 'right';
      td.style.paddingRight = '8px';
      td.textContent = q.strategies.p1[i];
      tr.insertBefore(td, tr.firstChild);
    }
  });
}

function showScore() {
  document.getElementById('quiz-area').style.display = 'none';
  const ss = document.getElementById('score-screen');
  ss.style.display = 'block';
  const pct = Math.round((state.correct / QUESTIONS.length) * 100);
  document.getElementById('final-pct').textContent = pct + '%';
  document.getElementById('final-correct').textContent = state.correct;
  document.getElementById('final-wrong').textContent = state.wrong;
  document.getElementById('final-pts').textContent = state.score;
  document.getElementById('progress').style.width = '100%';
}
